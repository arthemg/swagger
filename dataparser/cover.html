func getJSON(url string, target interface{}) error {

        var httpClient = &http.Client{Timeout: 10 * time.Second}

        r, err := httpClient.Get(url)

        if err != nil {
                return err
        }
        defer r.Body.Close()
        fmt.Println("our R", r.Body)
        return json.NewDecoder(r.Body).Decode(target)
}
